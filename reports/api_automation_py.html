14 def _make_request(self, endpoint: str, params: dict = None) -> Dict[Any, Any]: 

15 """發送 API 請求並處理錯誤""" 

16 try: 

17 url = f"{self.base_url}/{endpoint}" 

18 response = self.session.get(url, params=params, timeout=30) 

19 response.raise_for_status() 

20 data = response.json() 

21 

22 # 如果回傳的是陣列，包裝成標準格式 

23 if isinstance(data, list):

31 def get_all_pages(self, endpoint: str) -> List[Dict[Any, Any]]: 

32 """獲取所有分頁的資料""" 

33 all_results = [] 

34 page = 1 

35 

36 while True: 

37 try: 

38 data = self._make_request(endpoint, params={"page": page}) 

39 all_results.extend(data.get("results", [])) 

40 

41 if not data.get("next"): 

42 break 

43 

44 page += 1 

45 time.sleep(0.1) # 避免過於頻繁的請求 

46 

47 except Exception as e:

49 break 

50 

51 return all_results

133 """獲取馬力超過指定值的車輛""" 

134 vehicles = self._get_cached_data("vehicles") 

135 

136 high_power_vehicles = [] 

137 

138 for vehicle in vehicles: 

139 try: 

140 # 處理馬力資料（可能包含非數字字符） 

141 max_speed = vehicle.get("max_atmosphering_speed", "unknown") 

142 if max_speed == "unknown" or max_speed == "n/a": 

143 continue 

144 

145 # 移除非數字字符並轉換為整數

148 if speed_value > min_horsepower: 

149 high_power_vehicles.append({

153 "speed_value": speed_value

155 

156 except (ValueError, TypeError): 

157 continue 

158 

159 return sorted(high_power_vehicles, key=lambda x: x["speed_value"], reverse=True)
